## Installation
`npm install --save react-initicons`

## Usage
```JavaScript
var Initicon = require('react-initicon');

<Initicon
  size={100}
  text="Test Account"
  number={23}
  single={false}
/>

```

## React Native
```JavaScript
var Initicon = require('react-initicon/native');
```
